**（コード）**

```julia
# Pythonパッケージとの連携事例(PyCall)
using PyCall
@pyimport matplotlib.pyplot as plt
x = LinRange(0,2*π,1000); 
#y = sin(3*x + 4*cos(2*x))

len_x = length(x)
y = ones(len_x)

for i=1:len_x 
	y[i] = sin(3*x[i] + 4*cos(2*x[i]))
end

plt.plot(x, y, color="red", linewidth=2.0, linestyle="--")
plt.title("A sinusoidally modulated sinusoid")
plt.show()

```

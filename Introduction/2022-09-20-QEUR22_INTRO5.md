**（コード）**

```julia
# 3Dグラフを描く
using PyPlot
n = 100
u = LinRange(0,2*π,n)
v = LinRange(0,π,n)

len_u = length(u)
len_v = length(v)

x = ones(len_u, len_v)
y = ones(len_u, len_v)
z = ones(len_u, len_v)

for i=1:len_u 
    for j=1:len_v
        x[i,j] = sin(u[i])
        y[i,j] = cos(u[i])
        z[i,j] = sin(v[j])
    end
end

# The rstride and cstride arguments default to 10
surf(x,y,z, rstride=4, cstride=4)

# colored
colors = rand(len_u, len_v, 3)
fig = figure()
surf(x, y, z, facecolors=colors) 
fig[:canvas][:draw]()

```
